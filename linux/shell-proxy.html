Linux命令行实现代理连接上网

在profile文件或者.bashrc中设置相关环境变量即可。

1
2
3
4
5
6
$ vim ~/.bashrc
http_proxy=yourproxyaddress:proxyport   #分别指定http、https、ftp协议使用的代理服务器地址
https_proxy=yourproxyaddress:proxyport
ftp_proxy=yourproxyaddress:proxyport
no_proxy=192.168.10.0. #访问局域网地址（192.168.20.0/24网段）时不使用代理，可以用逗号分隔多个地址
export http_proxy https_proxy ftp_proxy no_proxy
